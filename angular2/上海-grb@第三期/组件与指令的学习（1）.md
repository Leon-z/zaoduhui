##组件指令的学习

###es5的用法
var App=ng.core.Component({
    selector:"app",
    template:'<h1>Hello,{target}</h1>'
})
.Class({
    constructor:function(){
       this.target="world";
    }
})

ng.platform.browser.bootstrap(App);

###typescript的用法
import {Component} from 'angular2/core';
import {bootstrap} from 'angular/platform/browser';

@Component({
    selector:"app",
    template:'<h1>Hello,{target}</h1>'
})
class App{
    target:string;
    constructor(){
      this.target="world
    }
}
bootstrap(App);

###angular2对指令的用法约定
+ propertyName="name"
+ [propertyName]="expression"
+ (eventName)="handler()"

###模板语法糖
+ 原始用法：

       <li>{{todo}}</li>
   </template>
+  语法糖
